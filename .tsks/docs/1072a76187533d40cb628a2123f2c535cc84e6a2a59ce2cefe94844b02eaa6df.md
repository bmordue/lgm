## 1. Summary

Integrate the combat mechanics (weapon range validation, damage calculation, attack orders) into the existing game loop and turn processing system. This task connects all combat system components to create a functioning combat experience within the turn-based strategy game.

## 2. Problem

The current game loop in `api/service/Rules.ts` has basic combat support but lacks:

- **Full Combat Integration**: Attack orders are processed but use simplified logic
- **Damage System**: No comprehensive damage calculation (just flat weapon damage)
- **Range Validation**: Basic range check exists but doesn't use hex distance properly
- **Line-of-Sight**: LOS checking exists but may not integrate with combat flow
- **Combat State**: No tracking of combat-specific state (cover, aim, overwatch)
- **Turn Results**: Combat outcomes not properly reported to players

### Current State
From `api/service/Rules.ts`:

```typescript
async function applyFiringRules(actorOrders: ActorOrders, game: Game, world: World, allActorsInWorld: Actor[], timestep: number): Promise<Actor> {
    // Basic implementation exists
    // - Checks orderType === OrderType.ATTACK
    // - Validates target exists and is alive
    // - Checks weapon range (using hex distance)
    // - Checks line of sight
    // - Applies flat weapon.damage to target health
    // - Handles death when health <= 0
}
```

Current limitations:
- Damage is flat `weapon.damage` value
- No damage modifiers (distance, terrain, cover, armor)
- No combat action types beyond ATTACK
- No combat state tracking
- Combat results not differentiated in turn results

### Impact
Without proper integration:
- Combat lacks strategic depth
- No tactical choices beyond "move and shoot"
- Turn results don't explain combat outcomes
- No foundation for advanced combat features

## 3. Proposed Solution

Integrate combat mechanics into the game loop by:

1. **Update Turn Processing**: Modify `processGameTurn()` to handle combat properly
2. **Combat Action Processing**: Process all combat action types (ATTACK, MELEE, etc.)
3. **Damage Application**: Use CombatMath for damage calculation
4. **State Tracking**: Track combat state (ammo, cover, aim)
5. **Turn Results**: Include combat details in turn results
6. **Event Logging**: Log combat events for UI display

import { Actor, ActorOrders, OrderType, Game, World, CombatEvent, CombatEventType } from './Models';
import { validateAttack, isTargetInRange } from './RangeValidation';
import { calculateDamage } from './CombatMath';
import { hasLineOfSight } from './Visibility';
import { Hex } from '../Hex';
import logger = require('../utils/Logger');
import util = require('util');

export interface CombatActionResult {
    success: boolean;
    message: string;
    events: CombatEvent[];
    stateChanges?: Partial<Actor>;
}

/**
 * Process combat action within game loop
 */
export async function processCombatAction(
    actor: Actor,
    order: ActorOrders,
    game: Game,
    world: World,
    allActors: Actor[],
    timestep: number
): Promise<CombatActionResult> {
    const events: CombatEvent[] = [];

    // Check actor is alive
    if (actor.state !== ActorState.ALIVE) {
        return {
            success: false,
            message: 'Actor is dead',
            events
        };
    }

    // Check actor hasn't already acted
    if (actor.hasActed) {
        return {
            success: false,
            message: 'Actor has already acted',
            events
        };
    }

    switch (order.orderType) {
        case OrderType.ATTACK:
            return await processAttack(actor, order, game, world, allActors, timestep);
        
        case OrderType.MOVE:
            // Movement handled separately
            return {
                success: true,
                message: 'Movement complete',
                events,
                stateChanges: { hasActed: true }
            };
        
        default:
            return {
                success: false,
                message: `Unknown order type: ${order.orderType}`,
                events
            };
    }
}

/**
 * Process attack action
 */
async function processAttack(
    attacker: Actor,
    order: ActorOrders,
    game: Game,
    world: World,
    allActors: Actor[],
    timestep: number
): Promise<CombatActionResult> {
    const events: CombatEvent[] = [];

    // Validate target
    if (!order.targetId) {
        return {
            success: false,
            message: 'No target specified',
            events
        };
    }

    const target = allActors.find(a => a.id === order.targetId);
    if (!target) {
        return {
            success: false,
            message: 'Target not found',
            events
        };
    }

    // Validate attack
    const validation = await validateAttack(attacker, target, world);
    if (!validation.valid) {
        return {
            success: false,
            message: `Invalid attack: ${validation.errors.join(', ')}`,
            events
        };
    }

    // Calculate damage
    const attackerTerrain = world.terrain[attacker.pos.x]?.[attacker.pos.y] || 0;
    const targetTerrain = world.terrain[target.pos.x]?.[target.pos.y] || 0;
    
    const damageCalc = calculateDamage(
        attacker,
        target,
        validation.distance,
        attackerTerrain,
        targetTerrain,
        validation.hasLineOfSight
    );

    // Apply damage
    target.health = Math.max(0, (target.health || 100) - damageCalc.finalDamage);

    // Consume ammo
    if (attacker.weapon?.ammo) {
        attacker.currentAmmo = (attacker.currentAmmo || attacker.weapon.ammo) - 1;
    }

    // Record attack event
    events.push({
        type: CombatEventType.ATTACK,
        actorId: attacker.id,
        targetId: target.id,
        damage: damageCalc.finalDamage,
        message: `Actor ${attacker.id} attacked Actor ${target.id} for ${damageCalc.finalDamage} damage`,
        timestep
    });

    // Check for kill
    if (target.health <= 0) {
        target.state = ActorState.DEAD;
        events.push({
            type: CombatEventType.UNIT_KILLED,
            actorId: attacker.id,
            targetId: target.id,
            message: `Actor ${target.id} was defeated`,
            timestep
        });
    }

    logger.info(util.format(
        'Combat: Actor %d -> %d: %d dmg (remaining HP: %d)',
        attacker.id, target.id, damageCalc.finalDamage, target.health
    ));

    return {
        success: true,
        message: `Attack dealt ${damageCalc.finalDamage} damage`,
        events,
        stateChanges: {
            hasActed: true,
            currentAmmo: attacker.currentAmmo
        }
    };
}

/**
 * Process overwatch triggers when actor moves
 */
export async function checkOverwatchTriggers(
    movingActor: Actor,
    fromPos: GridPosition,
    toPos: GridPosition,
    world: World,
    allActors: Actor[]
): Promise<CombatEvent[]> {
    const events: CombatEvent[] = [];

    const overwatchActors = allActors.filter(a =>
        a.isOnOverwatch &&
        a.state === ActorState.ALIVE &&
        a.weapon &&
        (a.currentAmmo || 0) > 0 &&
        a.owner !== movingActor.owner
    );

    for (const overwatchActor of overwatchActors) {
        if (overwatchActor.overwatchTriggered) continue;

        const overwatchHex = gridPositionToHex(overwatchActor.pos);
        const movingHex = gridPositionToHex(movingActor.pos);

        const hasLOS = hasLineOfSight(overwatchHex, movingHex, world.terrain, allActors);
        if (!hasLOS) continue;

        const rangeCheck = isTargetInRange(
            overwatchActor.pos,
            movingActor.pos,
            overwatchActor.weapon!
        );

        if (!rangeCheck.inRange) continue;

        // Trigger overwatch fire
        const damageCalc = calculateDamage(
            overwatchActor,
            movingActor,
            rangeCheck.distance,
            world.terrain[overwatchActor.pos.x]?.[overwatchActor.pos.y] || 0,
            world.terrain[movingActor.pos.x]?.[movingActor.pos.y] || 0,
            true
        );

        // Overwatch penalty: -30% damage
        const overwatchDamage = Math.floor(damageCalc.finalDamage * 0.7);
        movingActor.health = Math.max(0, (movingActor.health || 100) - overwatchDamage);

        overwatchActor.currentAmmo! -= 1;
        overwatchActor.overwatchTriggered = true;

        events.push({
            type: CombatEventType.OVERWATCH_FIRE,
            actorId: overwatchActor.id,
            targetId: movingActor.id,
            damage: overwatchDamage,
            message: `Overwatch: Actor ${overwatchActor.id} fired at ${movingActor.id} for ${overwatchDamage} damage`,
            timestep: 0
        });

        if (movingActor.health <= 0) {
            movingActor.state = ActorState.DEAD;
            events.push({
                type: CombatEventType.UNIT_KILLED,
                actorId: overwatchActor.id,
                targetId: movingActor.id,
                message: `Actor ${movingActor.id} was defeated by overwatch`,
                timestep: 0
            });
        }
    }

    return events;
}

function gridPositionToHex(pos: GridPosition): Hex {
    const q = pos.y;
    const r = pos.x - (pos.y - (pos.y & 1)) / 2;
    return new Hex(q, r, -q - r);
}

import { processCombatAction, checkOverwatchTriggers } from './CombatIntegrationService';

export async function applyRulesToActorOrders(
    game: Game,
    world: World,
    allActorOrders: Array<ActorOrders>,
    allActorsInWorld: Actor[]
): Promise<Array<Actor>> {
    if (!allActorOrders || allActorOrders.length === 0) {
        logger.warn("applyRulesToActorOrders: did not receive any orders");
        return allActorsInWorld;
    }

    // Reset acted status for new turn
    for (const actor of allActorsInWorld) {
        actor.hasActed = false;
        actor.actionPoints = actor.actionPoints || 1;
    }

    const allCombatEvents: CombatEvent[] = [];

    // Process orders by timestep
    for (let ts = 0; ts < TIMESTEP_MAX; ts++) {
        for (const ao of allActorOrders) {
            const actor = allActorsInWorld.find(a => a.id === ao.actorId);
            if (!actor || actor.state !== ActorState.ALIVE) continue;

            // Handle movement with overwatch triggers
            if (ao.orderType === OrderType.MOVE && ao.ordersList) {
                for (const direction of ao.ordersList) {
                    const fromPos = { ...actor.pos };
                    const newPos = applyDirection(actor.pos, direction);
                    
                    // Check overwatch triggers
                    const overwatchEvents = await checkOverwatchTriggers(
                        actor,
                        fromPos,
                        newPos,
                        world,
                        allActorsInWorld
                    );
                    allCombatEvents.push(...overwatchEvents);

                    // Apply movement if still alive
                    if (actor.state === ActorState.ALIVE) {
                        await applyMovementOrders(ao, game, world, ts, allActorsInWorld);
                    }
                }
            }

            // Handle combat actions
            if (ao.orderType === OrderType.ATTACK) {
                const result = await processCombatAction(
                    actor,
                    ao,
                    game,
                    world,
                    allActorsInWorld,
                    ts
                );

                if (result.success && result.stateChanges) {
                    Object.assign(actor, result.stateChanges);
                }

                allCombatEvents.push(...result.events);

                logger.debug(util.format(
                    'Combat action %s: %s',
                    OrderType[ao.orderType],
                    result.message
                ));
            }
        }
    }

    // Reset temporary combat states
    for (const actor of allActorsInWorld) {
        if (actor.isAiming) {
            actor.isAiming = false;
            actor.aimBonus = 0;
        }
        if (actor.isOnOverwatch) {
            actor.isOnOverwatch = false;
            actor.overwatchTriggered = false;
        }
    }

    // Store combat events for turn results
    (game as any).combatEvents = allCombatEvents;

    return allActorsInWorld;
}

function turnResultsPerPlayer(
    game: Game,
    allUpdatedActorsThisTurn: Array<Actor>,
    combatEvents: CombatEvent[]
): Array<TurnResult> {
    if (!currentTurnWorldTerrain) {
        logger.error("turnResultsPerPlayer: currentTurnWorldTerrain is not set");
        return [];
    }

    const fullWorldStateForThisTurn: { terrain: number[][], actors: Actor[] } = {
        actors: allUpdatedActorsThisTurn.map(a => ({ ...a })),
        terrain: currentTurnWorldTerrain
    };

    return game.players.map((playerId) => {
        const visibleWorld = getVisibleWorldForPlayer(fullWorldStateForThisTurn, playerId);

        // Filter combat events for this player
        const playerEvents = combatEvents.filter(event => {
            const actor = allUpdatedActorsThisTurn.find(a => a.id === event.actorId);
            const target = event.targetId ? allUpdatedActorsThisTurn.find(a => a.id === event.targetId) : undefined;
            return actor?.owner === playerId || target?.owner === playerId;
        });

        return {
            gameId: game.id,
            playerId: playerId,
            turn: game.turn,
            world: {
                terrain: visibleWorld.terrain,
                actorIds: visibleWorld.actorIds,
                actors: visibleWorld.actors
            },
            combatEvents: playerEvents  // NEW
        };
    });
}
