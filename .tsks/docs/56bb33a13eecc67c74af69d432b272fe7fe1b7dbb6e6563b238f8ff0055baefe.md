- **Partial Coverage**: Only some components tested

## 3. Proposed Solution

1. **Unit Tests**: Test each component in isolation

### Phase 2: HexGrid Component Tests (3-4 hours)

Create/update `client/src/components/__tests__/HexGrid.spec.ts`:
```typescript
import { describe, it, expect } from 'vitest';
import { mount } from '@vue/test-utils';
import HexGrid from '../HexGrid.vue';
import { createPinia } from 'pinia';

describe('HexGrid', () => {
    const createWrapper = (props = {}) => {
        const pinia = createPinia();
        return mount(HexGrid, {
            props,
            global: {
                plugins: [pinia]
            }
        });
    };

    it('should render grid with correct dimensions', () => {
        const wrapper = createWrapper({
            width: 10,
            height: 10
        });
        
        expect(wrapper.findAll('.hex-cell')).toHaveLength(100);
    });

    it('should display actors on correct hexes', async () => {
        const wrapper = createWrapper({
            actors: [
                { id: 1, pos: { x: 5, y: 5 }, owner: 1 }
            ]
        });
        
        const actorToken = wrapper.findComponent({ name: 'ActorToken' });
        expect(actorToken.exists()).toBe(true);
    });

    it('should emit hex-click event', async () => {
        const wrapper = createWrapper();
        const hexCell = wrapper.find('.hex-cell');
        
        await hexCell.trigger('click');
        
        expect(wrapper.emitted('hex-click')).toBeDefined();
    });

    it('should highlight selected actor', async () => {
        const wrapper = createWrapper({
            selectedActorId: 1
        });
        
        const selectedHex = wrapper.find('.hex-cell.selected');
        expect(selectedHex.exists()).toBe(true);
    });
});
```

### Phase 3: ActorToken Component Tests (2-3 hours)

Create `client/src/components/__tests__/ActorToken.spec.ts`:
```typescript
import { describe, it, expect } from 'vitest';
import { mount } from '@vue/test-utils';
import ActorToken from '../ActorToken.vue';

describe('ActorToken', () => {
    it('should render with owner color', () => {
        const wrapper = mount(ActorToken, {
            props: {
                actor: { id: 1, owner: 1, health: 100, maxHealth: 100 }
            }
        });
        
        expect(wrapper.classes()).toContain('owner-1');
    });

    it('should display health bar', () => {
        const wrapper = mount(ActorToken, {
            props: {
                actor: { id: 1, owner: 1, health: 50, maxHealth: 100 },
                showHealth: true
            }
        });
        
        const healthFill = wrapper.find('.health-fill');
        expect(healthFill.exists()).toBe(true);
        expect(healthFill.attributes('style')).toContain('width: 50%');
    });

    it('should not show health when disabled', () => {
        const wrapper = mount(ActorToken, {
            props: {
                actor: { id: 1, owner: 1, health: 50, maxHealth: 100 },
                showHealth: false
            }
        });
        
        expect(wrapper.find('.health-bar').exists()).toBe(false);
    });
});
```

### Phase 4: View Component Tests (3-4 hours)

Create `client/src/views/__tests__/GameView.spec.ts`:
```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import { mount } from '@vue/test-utils';
import { createPinia, setActivePinia } from 'pinia';
import GameView from '../GameView.vue';
import { useGamesStore } from '@/stores/Games.store';

describe('GameView', () => {
    let pinia: ReturnType<typeof createPinia>;
    let gamesStore: ReturnType<typeof useGamesStore>;

    beforeEach(() => {
        pinia = createPinia();
        setActivePinia(pinia);
        gamesStore = useGamesStore();
    });

    it('should render game view', () => {
        const wrapper = mount(GameView, {
            global: {
                plugins: [pinia]
            }
        });
        
        expect(wrapper.find('.game-view').exists()).toBe(true);
    });

    it('should display current turn', async () => {
        gamesStore.currentGame = { turn: 5 };
        
        const wrapper = mount(GameView, {
            global: {
                plugins: [pinia]
            }
        });
        
        expect(wrapper.text()).toContain('Turn 5');
    });

    it('should show order submission form', () => {
        const wrapper = mount(GameView, {
            global: {
                plugins: [pinia]
            }
        });
        
        expect(wrapper.findComponent({ name: 'OrderSubmission' }).exists()).toBe(true);
    });
});
```
