    for (let x = 0; x < grid.length; x++) {

            grid[x][y] = defaultValue;
        }

import { describe, it, expect } from 'vitest';
import { drawLine, drawCircle, drawRectangle } from '../utils/Draw';

describe('Draw Utility', () => {
    describe('drawLine', () => {
        it('should draw horizontal line', () => {
            const grid = createGrid(10, 10);
            drawLine(grid, { x: 0, y: 5 }, { x: 9, y: 5 });
            
            for (let x = 0; x < 10; x++) {
                expect(grid[x][5]).toBe(1);
            }
        });

        it('should draw vertical line', () => {
            const grid = createGrid(10, 10);
            drawLine(grid, { x: 5, y: 0 }, { x: 5, y: 9 });
            
            for (let y = 0; y < 10; y++) {
                expect(grid[5][y]).toBe(1);
            }
        });

        it('should draw diagonal line', () => {
            const grid = createGrid(10, 10);
            drawLine(grid, { x: 0, y: 0 }, { x: 9, y: 9 });
            
            for (let i = 0; i < 10; i++) {
                expect(grid[i][i]).toBe(1);
            }
        });
    });

    describe('drawCircle', () => {
        it('should draw circle', () => {
            const grid = createGrid(20, 20);
            drawCircle(grid, { centerX: 10, centerY: 10, radius: 5 });
            
            // Verify circle points
        });
    });

    describe('drawRectangle', () => {
        it('should draw rectangle outline', () => {
            const grid = createGrid(20, 20);
            drawRectangle(grid, { x: 5, y: 5, width: 10, height: 10 });
            
            // Verify rectangle points
        });
    });
});

function createGrid(width: number, height: number): any[][] {
    return Array(width).fill(null).map(() => Array(height).fill(0));
}
