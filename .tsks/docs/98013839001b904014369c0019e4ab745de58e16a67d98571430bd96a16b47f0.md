# Implementation Plan: Create Combat System Tests

**Issue ID**: lgm-7eg
**Priority**: P2
**Type**: Task
**Depends On**: 
- lgm-vfj (Add combat-related unit actions)
- lgm-9u0 (Integrate combat mechanics into game loop)
- lgm-sqo (Implement damage calculation formulas)
- lgm-m0a (Implement weapon range validation logic)

## 1. Summary

Create comprehensive unit tests and integration tests for the combat system, covering line-of-sight calculations, weapon range validation, damage calculation formulas, attack order processing, and combat integration with the game loop.

## 2. Problem

The combat system implementation (lgm-9u0, lgm-sqo, lgm-m0a, lgm-vfj) adds complex mechanics including:

- **Damage Calculations**: Multiple factors affect damage (weapon stats, distance, terrain, cover, armor)
- **Range Validation**: Min/max range checks, hex distance calculations
- **Line-of-Sight**: Complex pathfinding and occlusion checks
- **Attack Orders**: Processing ATTACK order types in turn resolution
- **Combat State**: Actor health, death, and state transitions

Without comprehensive tests:
- **Bugs in Combat Logic**: Incorrect damage, range, or LOS calculations
- **Regression Risk**: Changes break existing combat functionality
- **Unclear Behavior**: Tests document expected combat system behavior
- **Integration Issues**: Combat doesn't work correctly with game loop

### Current State
From examining the codebase:
- `api/test/` contains some test files (e2e.test.ts, GameService.test.ts, etc.)
- `api/service/Rules.ts` has combat logic in `applyFiringRules`
- `api/service/Visibility.ts` has `hasLineOfSight` function
- `api/config/WeaponsConfig.ts` defines weapon types
- No dedicated combat system test suite exists

### Testing Gaps
- No unit tests for damage calculation (lgm-sqo)
- No unit tests for range validation (lgm-m0a)
- No integration tests for combat in game loop
- No tests for line-of-sight with various terrain configurations
- No tests for edge cases (self-attack, dead targets, etc.)

## 3. Proposed Solution

Create a comprehensive test suite covering all combat system components:

1. **Unit Tests**: Test individual functions in isolation
2. **Integration Tests**: Test combat within game loop
3. **Edge Case Tests**: Test boundary conditions and error scenarios
4. **Performance Tests**: Ensure combat calculations don't slow down turns

import { Actor, ActorState, Terrain, World, GridPosition, Weapon } from '../../../service/Models';
import { WEAPON_TYPES, WeaponDefinition } from '../../../config/WeaponsConfig';

export interface CombatTestSetup {
    attacker: Actor;
    defender: Actor;
    world: World;
    allActors: Actor[];
}

/**
 * Create a test actor with specified properties
 */
export function createTestActor(
    id: number,
    pos: GridPosition,
    owner: number,
    weaponId: string = 'STANDARD_BLASTER',
    overrides: Partial<Actor> = {}
): Actor {
    const weapon = { ...WEAPON_TYPES[weaponId] };
    
    return {
        id,
        pos: { ...pos },
        state: ActorState.ALIVE,
        owner,
        health: 100,
        maxHealth: 100,
        weapon,
        armor: 0,
        ...overrides
    };
}

/**
 * Create empty terrain grid
 */
export function createEmptyTerrain(width: number, height: number): Terrain[][] {
    return Array(width).fill(null).map(() =>
        Array(height).fill(Terrain.EMPTY)
    );
}

/**
 * Create terrain with blocked cells
 */
export function createTerrainWithBlocks(
    width: number,
    height: number,
    blockedPositions: GridPosition[]
): Terrain[][] {
    const terrain = createEmptyTerrain(width, height);
    
    for (const pos of blockedPositions) {
        if (pos.x >= 0 && pos.x < width && pos.y >= 0 && pos.y < height) {
            terrain[pos.x][pos.y] = Terrain.BLOCKED;
        }
    }
    
    return terrain;
}

/**
 * Create a test world
 */
export function createTestWorld(terrain: Terrain[][], actorIds: number[] = []): World {
    return {
        id: 1,
        terrain,
        actorIds,
        actors: []
    };
}

/**
 * Setup basic combat scenario
 */
export function setupBasicCombat(): CombatTestSetup {
    const terrain = createEmptyTerrain(20, 20);
    
    const attacker = createTestActor(1, { x: 5, y: 5 }, 1, 'RIFLE');
    const defender = createTestActor(2, { x: 10, y: 5 }, 2, 'PISTOL');
    
    const world = createTestWorld(terrain, [1, 2]);
    const allActors = [attacker, defender];
    
    return { attacker, defender, world, allActors };
}

/**
 * Assert damage calculation is within expected range
 */
export function expectDamageInRange(
    damage: number,
    min: number,
    max: number,
    message?: string
) {
    if (damage < min || damage > max) {
        throw new Error(
            message || `Damage ${damage} not in range [${min}, ${max}]`
        );
    }
}

import { expect } from 'chai';
import * as RangeValidation from '../../service/RangeValidation';

            expect(distance).to.equal(1);
        });

            const pos2 = { x: 3, y: 4 };
            const distance = RangeValidation.calculateHexDistance(pos1, pos2);
            expect(distance).to.be.gte(4); // Hex distance
        });

        it('should be symmetric (distance A->B = B->A)', () => {
            const pos1 = { x: 2, y: 7 };
            const pos2 = { x: 8, y: 3 };
            const dist1 = RangeValidation.calculateHexDistance(pos1, pos2);
            const dist2 = RangeValidation.calculateHexDistance(pos2, pos1);
            expect(dist1).to.equal(dist2);
        });
    });

    describe('isTargetInRange', () => {
        const rifle = { ...WEAPON_TYPES.RIFLE }; // minRange: 1, maxRange: 8
        
        it('should return true for target at optimal range', () => {
            const result = RangeValidation.isTargetInRange(
                { x: 0, y: 0 },
                { x: 5, y: 0 },
                rifle
            );
            expect(result.inRange).to.be.true;
            expect(result.distance).to.equal(5);
        });

        it('should return false for target too close (min range violation)', () => {
            const sniper = { ...WEAPON_TYPES.SNIPER }; // minRange: 5
            const result = RangeValidation.isTargetInRange(
                { x: 0, y: 0 },
                { x: 2, y: 0 },
                sniper
            );
            expect(result.inRange).to.be.false;
            expect(result.tooClose).to.be.true;
            expect(result.reason).to.include('too close');
        });

        it('should return false for target too far (max range exceeded)', () => {
            const pistol = { ...WEAPON_TYPES.PISTOL }; // maxRange: 3
            const result = RangeValidation.isTargetInRange(
                { x: 0, y: 0 },
                { x: 10, y: 0 },
                pistol
            );
            expect(result.inRange).to.be.false;
            expect(result.tooFar).to.be.true;
            expect(result.reason).to.include('too far');
        });

        it('should handle melee weapons (minRange: 0)', () => {
            const melee = { ...WEAPON_TYPES.MELEE }; // minRange: 0, maxRange: 1
            const adjacent = RangeValidation.isTargetInRange(
                { x: 0, y: 0 },
                { x: 1, y: 0 },
                melee
            );
            expect(adjacent.inRange).to.be.true;

            const samePosition = RangeValidation.isTargetInRange(
                { x: 0, y: 0 },
                { x: 0, y: 0 },
                melee
            );
            expect(samePosition.inRange).to.be.true;
        });
    });

    describe('canAttackTarget', () => {
        it('should return true for valid attack', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'RIFLE');
            const target = createTestActor(2, { x: 5, y: 0 }, 2, 'PISTOL');
            
            const canAttack = RangeValidation.canAttackTarget(attacker, target);
            expect(canAttack).to.be.true;
        });

        it('should return false for actor without weapon', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'STANDARD_BLASTER');
            attacker.weapon = undefined;
            
            const target = createTestActor(2, { x: 5, y: 0 }, 2);
            
            const canAttack = RangeValidation.canAttackTarget(attacker, target);
            expect(canAttack).to.be.false;
        });

        it('should return false for out-of-range target', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'PISTOL');
            const target = createTestActor(2, { x: 15, y: 0 }, 2);
            
            const canAttack = RangeValidation.canAttackTarget(attacker, target);
            expect(canAttack).to.be.false;
        });
    });

    describe('getValidTargets', () => {
        it('should return only enemies in range', async () => {
            const attacker = createTestActor(1, { x: 5, y: 5 }, 1, 'RIFLE');
            const inRangeEnemy = createTestActor(2, { x: 8, y: 5 }, 2);
            const outOfRangeEnemy = createTestActor(3, { x: 20, y: 5 }, 2);
            const friendly = createTestActor(4, { x: 7, y: 5 }, 1);
            
            const allActors = [attacker, inRangeEnemy, outOfRangeEnemy, friendly];
            const terrain = createEmptyTerrain(25, 25);
            
            const targets = await RangeValidation.getValidTargets(
                attacker,
                allActors,
                terrain
            );
            
            expect(targets).to.have.length(1);
            expect(targets[0].id).to.equal(inRangeEnemy.id);
        });

        it('should exclude dead actors', async () => {

import { expect } from 'chai';
import * as CombatMath from '../../service/CombatMath';
import { Actor, ActorState, Terrain } from '../../service/Models';
import { WEAPON_TYPES } from '../../config/WeaponsConfig';
import { createTestActor, createEmptyTerrain, createTerrainWithBlocks } from './helpers/combatTestHelpers';

describe('Combat System - Damage Calculation', () => {
    describe('calculateDamage - Base Damage', () => {
        it('should calculate base damage at optimal range', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'RIFLE');
            const defender = createTestActor(2, { x: 5, y: 0 }, 2);
            const terrain = createEmptyTerrain(20, 20);
            
            const damage = CombatMath.calculateDamage(
                attacker,
                defender,
                5,  // Optimal range for rifle
                terrain[0][0],
                terrain[5][0],
                true
            );
            
            expect(damage.finalDamage).to.be.gt(0);
            expect(damage.baseDamage).to.equal(attacker.weapon!.baseDamage);
        });

        it('should deal minimum 1 damage on hit', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'PISTOL');
            const defender = createTestActor(2, { x: 2, y: 0 }, 2, 'PISTOL', {
                armor: 50  // Max armor
            });
            const terrain = createEmptyTerrain(20, 20);
            
            const damage = CombatMath.calculateDamage(
                attacker,
                defender,
                2,
                terrain[0][0],
                terrain[2][0],
                true
            );
            
            if (damage.finalDamage > 0) {
                expect(damage.finalDamage).to.be.gte(1);
            }
        });

        it('should return 0 damage for no weapon', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1);
            attacker.weapon = undefined;
            
            const defender = createTestActor(2, { x: 2, y: 0 }, 2);
            const terrain = createEmptyTerrain(20, 20);
            
            const damage = CombatMath.calculateDamage(
                attacker,
                defender,
                2,
                terrain[0][0],
                terrain[2][0],
                true
            );
            
            expect(damage.finalDamage).to.equal(0);
            expect(damage.breakdown).to.include('no weapon');
        });
    });

    describe('calculateDamage - Distance Modifiers', () => {
        it('should reduce damage beyond optimal range', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'RIFLE');
            const defender = createTestActor(2, { x: 8, y: 0 }, 2);
            const terrain = createEmptyTerrain(20, 20);
            
            const closeDamage = CombatMath.calculateDamage(
                attacker,
                defender,
                5,  // Optimal
                terrain[0][0],
                terrain[5][0],
                true
            );
            
            const farDamage = CombatMath.calculateDamage(
                attacker,
                defender,
                8,  // Max range
                terrain[0][0],
                terrain[8][0],
                true
            );
            
            expect(farDamage.finalDamage).to.be.lt(closeDamage.finalDamage);
        });

        it('should return 0 damage for out-of-range attacks', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'PISTOL');
            const defender = createTestActor(2, { x: 10, y: 0 }, 2);
            const terrain = createEmptyTerrain(20, 20);
            
            const damage = CombatMath.calculateDamage(
                attacker,
                defender,
                10,  // Beyond pistol maxRange (3)
                terrain[0][0],
                terrain[10][0],
                true
            );
            
            expect(damage.finalDamage).to.equal(0);
            expect(damage.breakdown).to.include('out of range');
        });

        it('should apply min range penalty', () => {
            const sniper = createTestActor(1, { x: 0, y: 0 }, 1, 'SNIPER');
            const target = createTestActor(2, { x: 2, y: 0 }, 2);
            const terrain = createEmptyTerrain(20, 20);
            
            const damage = CombatMath.calculateDamage(
                sniper,
                target,
                2,  // Within minRange (5)
                terrain[0][0],
                terrain[2][0],
                true
            );
            
            expect(damage.finalDamage).to.equal(0);
        });
    });

    describe('calculateDamage - Terrain and Cover', () => {
        it('should reduce damage for defender in cover', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'RIFLE');
            const defender = createTestActor(2, { x: 5, y: 0 }, 2);
            
            const noCoverDamage = CombatMath.calculateDamage(
                attacker,
                defender,
                5,
                Terrain.EMPTY,
                Terrain.EMPTY,
                true
            );
            
            const coverDamage = CombatMath.calculateDamage(
                attacker,
                defender,
                5,
                Terrain.EMPTY,
                Terrain.BLOCKED,  // Full cover
                true
            );
            
            expect(coverDamage.finalDamage).to.be.lt(noCoverDamage.finalDamage);
            expect(coverDamage.coverModifier).to.be.lt(1.0);
        });

        it('should apply terrain accuracy modifiers', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'RIFLE');
            const defender = createTestActor(2, { x: 5, y: 0 }, 2);
            
            const clearDamage = CombatMath.calculateDamage(
                attacker,
                defender,
                5,
                Terrain.EMPTY,
                Terrain.EMPTY,
                true
            );
            
            // Attacker in difficult terrain should have reduced accuracy
            const difficultTerrainDamage = CombatMath.calculateDamage(
                attacker,
                defender,
                5,
                Terrain.BLOCKED,  // Difficult terrain for attacker
                Terrain.EMPTY,
                true
            );
            
            expect(difficultTerrainDamage.finalDamage).to.be.lte(clearDamage.finalDamage);
        });
    });

    describe('calculateDamage - Armor', () => {
        it('should reduce damage based on armor', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'PISTOL');
            const noArmor = createTestActor(2, { x: 2, y: 0 }, 2);
            const heavyArmor = createTestActor(2, { x: 2, y: 0 }, 2, 'PISTOL', {
                armor: 40
            });
            const terrain = createEmptyTerrain(20, 20);
            
            const noArmorDamage = CombatMath.calculateDamage(
                attacker,
                noArmor,
                2,
                terrain[0][0],
                terrain[2][0],
                true
            );
            
            const armorDamage = CombatMath.calculateDamage(
                attacker,
                heavyArmor,
                2,
                terrain[0][0],
                terrain[2][0],
                true
            );
            
            expect(armorDamage.finalDamage).to.be.lt(noArmorDamage.finalDamage);
        });

        it('should account for armor penetration', () => {
            const attacker = createTestActor(1, { x: 0, y: 0 }, 1, 'RIFLE');
